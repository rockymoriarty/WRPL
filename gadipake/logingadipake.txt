    <script>
        document.addEventListener("DOMContentLoaded", function () {
            const inputs = document.querySelectorAll(".form-container input");
            const button = document.querySelector(".form-container button");
        
            // Animasi input saat diklik
            inputs.forEach(input => {
                input.addEventListener("focus", () => {
                    input.style.border = "2px solid #8E2DE2";
                    input.style.transition = "0.3s";
                });
        
                input.addEventListener("blur", () => {
                    input.style.border = "none";
                });
            });
        
            // Validasi sederhana saat submit
            button.addEventListener("click", (event) => {
                event.preventDefault();
                const email = document.querySelector("input[name='email']").value;
                const password = document.querySelector("input[name='password']").value;
        
                if (email === "" || password === "") {
                    alert("Please fill in all fields!");
                } else {
                    button.innerHTML = "Logging in...";
                    setTimeout(() => {
                        alert("Login successful!");
                        button.innerHTML = "Login";
                        document.querySelector("form").reset();
                    }, 1500);
                }
            });
        
            // Efek hover tombol login
            button.addEventListener("mouseover", () => {
                button.style.background = "#8E2DE2";
                button.style.transition = "0.3s";
            });
        
            button.addEventListener("mouseout", () => {
                button.style.background = "#4A00E0";
            });
        });
        </script>
